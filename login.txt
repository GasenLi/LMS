    <script type="text/javascript">
        $(document).ready(function () {
            // 登录按钮事件
            $("#loginButton").click(function () {
                $("#loginForm").ajaxSubmit({
                    url: "./login.do",
                    type:"POST",
                    dataType:"json",
                    error:function(data){
                        swal({
                            title: "服务器错误，请联系管理员",
                            text: "5秒后自动消失",
                            timer: 5000,
                            type:"error",
                            showConfirmButton: true
                        });
                    },
                    success:function(data){
                        if (data == '1'){
                            self.location='pages/categoryManage/categoryManage.html';
                        }else if (data == "0"){
                            swal({
                                title: "用户不存在",
                                text: "5秒后自动消失",
                                timer: 5000,
                                type:"error",
                                showConfirmButton: true
                            });
                        }else if(data == '-1'){
                            swal({
                                title: "密码错误",
                                text: "5秒后自动消失",
                                timer: 5000,
                                type:"error",
                                showConfirmButton: true
                            });
                        }
                    }
                });
            });

            // 注册按钮事件
            $("#registerButton").click(function () {
                self.location='register.html';
            });
        });
    </script>
